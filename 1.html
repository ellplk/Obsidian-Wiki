<script>

  const canvas = document.getElementById('gameCanvas');

  const ctx = canvas.getContext('2d');

  const scoreEl = document.getElementById('score');


  // 游戏常量

  const GRID_SIZE = 20;

  const TILE_COUNT = canvas.width / GRID_SIZE;


  // 蛇：数组存储每个身体段的 {x, y}

  let snake = [{ x: 10, y: 10 }];

  let direction = { x: 0, y: 0 };

  let nextDirection = { x: 0, y: 0 };


  // 食物

  let food = {};


  // 状态

  let score = 0;

  let gameRunning = false;

  let gameSpeed = 120; // 毫秒每帧

  let gameLoop = null;

  let paused = false;


  // 随机生成食物位置

  function generateFood() {

    food = {

      x: Math.floor(Math.random() * TILE_COUNT),

      y: Math.floor(Math.random() * TILE_COUNT)

    };

    // 确保食物不生成在蛇身上

    for (let segment of snake) {

      if (segment.x === food.x && segment.y === food.y) {

        return generateFood();

      }

    }

  }


  // 绘制函数

  function draw() {

    // 清空画布

    ctx.fillStyle = '#16213e';

    ctx.fillRect(0, 0, canvas.width, canvas.height);


    // 绘制网格（可选视觉效果）

    ctx.strokeStyle = 'rgba(255,255,255,0.03)';

    ctx.lineWidth = 1;

    for (let i = 0; i <= TILE_COUNT; i++) {

      ctx.beginPath();

      ctx.moveTo(i * GRID_SIZE, 0);

      ctx.lineTo(i * GRID_SIZE, canvas.height);

      ctx.stroke();

    }

    for (let i = 0; i <= TILE_COUNT; i++) {

      ctx.beginPath();

      ctx.moveTo(0, i * GRID_SIZE);

      ctx.lineTo(canvas.width, i * GRID_SIZE);

      ctx.stroke();

    }


    // 绘制蛇

    snake.forEach((segment, index) => {

      // 蛇头颜色更深

      if (index === 0) {

        ctx.fillStyle = '#e94560';

      } else {

        // 身体由深到浅渐变

        const alpha = 0.4 + (index / snake.length) * 0.4;

        ctx.fillStyle = `rgba(233, 69, 96, ${alpha})`;

      }

      ctx.fillRect(

        segment.x * GRID_SIZE + 1,

        segment.y * GRID_SIZE + 1,

        GRID_SIZE - 2,

        GRID_SIZE - 2

      );


      // 蛇头眼睛细节

      if (index === 0) {

        ctx.fillStyle = '#fff';

        const eyeSize = 3;

        // 根据方向确定眼睛位置

        let eye1, eye2;

        if (direction.x === 1) { // 右

          eye1 = { x: segment.x + 5, y: segment.y + 5 };

          eye2 = { x: segment.x + 5, y: segment.y + 13 };

        } else if (direction.x === -1) { // 左

          eye1 = { x: segment.x + 2, y: segment.y + 5 };

          eye2 = { x: segment.x + 2, y: segment.y + 13 };

        } else if (direction.y === 1) { // 下

          eye1 = { x: segment.x + 5, y: segment.y + 11 };

          eye2 = { x: segment.x + 13, y: segment.y + 11 };

        } else if (direction.y === -1) { // 上

          eye1 = { x: segment.x + 5, y: segment.y + 2 };

          eye2 = { x: segment.x + 13, y: segment.y + 2 };

        } else { // 默认向右

          eye1 = { x: segment.x + 5, y: segment.y + 5 };

          eye2 = { x: segment.x + 5, y: segment.y + 13 };

        }

        ctx.fillRect(eye1.x, eye1.y, eyeSize, eyeSize);

        ctx.fillRect(eye2.x, eye2.y, eyeSize, eyeSize);

      }

    });


    // 绘制食物

    ctx.fillStyle = '#ff6b81';

    ctx.shadowColor = '#ff6b81';

    ctx.shadowBlur = 15;

    ctx.fillRect(

      food.x * GRID_SIZE + 3,

      food.y * GRID_SIZE + 3,

      GRID_SIZE - 6,

      GRID_SIZE - 6

    );

    ctx.shadowBlur = 0;


    // 绘制得分文字

    ctx.fillStyle = '#fff';

    ctx.font = '16px Arial';

    ctx.textAlign = 'center';

    ctx.fillText(`得分: ${score}`, canvas.width / 2, 30);

  }


  // 更新蛇的位置

  function update() {

    if (!gameRunning || paused) return;


    // 确保蛇不会反向移动

    if (nextDirection.x !== 0 || nextDirection.y !== 0) {

      if (direction.x + nextDirection.x !== 0 || direction.y + nextDirection.y !== 0) {

        direction = { ...nextDirection };

      }

    }


    // 计算蛇头新位置

    const head = {

      x: snake[0].x + direction.x,

      y: snake[0].y + direction.y

    };


    // 边界检查（循环穿越）

    if (head.x < 0) head.x = TILE_COUNT - 1;

    if (head.x >= TILE_COUNT) head.x = 0;

    if (head.y < 0) head.y = TILE_COUNT - 1;

    if (head.y >= TILE_COUNT) head.y = 0;


    // 撞到自己（不能吃到食物时）

    for (let i = 0; i < snake.length; i++) {

      if (head.x === snake[i].x && head.y === snake[i].y) {

        gameOver();

        return;

      }

    }


    snake.unshift(head);


    // 吃食物

    if (head.x === food.x && head.y === food.y) {

      score += 10;

      scoreEl.textContent = score;

      generateFood();

      // 加快速度，上限

      if (gameSpeed > 60) {

        gameSpeed -= 1.5;

        clearInterval(gameLoop);

        gameLoop = setInterval(gameLoopHandler, gameSpeed);

      }

    } else {

      snake.pop(); // 没吃到就移除尾部

    }

  }


  function gameLoopHandler() {

    update();

    draw();

  }


  function startGame() {

    // 重置状态

    snake = [{ x: 10, y: 10 }];

    direction = { x: 0, y: 0 };

    nextDirection = { x: 0, y: 0 };

    score = 0;

    gameSpeed = 120;

    scoreEl.textContent = '0';


    document.getElementById('startModal').style.display = 'none';

    document.getElementById('gameOverModal').style.display = 'none';


    generateFood();


    if (gameLoop) clearInterval(gameLoop);

    gameLoop = setInterval(gameLoopHandler, gameSpeed);

    gameRunning = true;

    paused = false;

  }


  function gameOver() {

    gameRunning = false;

    clearInterval(gameLoop);

    document.getElementById('finalScore').textContent = score;

    document.getElementById('gameOverModal').style.display = 'flex';

  }


  // 事件监听

  document.addEventListener('keydown', (e) => {

    if (!gameRunning) return;


    switch (e.key) {

      case 'ArrowUp':

      case 'w':

      case 'W':

        nextDirection = { x: 0, y: -1 };

        break;

      case 'ArrowDown':

      case 's':

      case 'S':

        nextDirection = { x: 0, y: 1 };

        break;

      case 'ArrowLeft':

      case 'a':

      case 'A':

        nextDirection = { x: -1, y: 0 };

        break;

      case 'ArrowRight':

      case 'd':

      case 'D':

        nextDirection = { x: 1, y: 0 };

        break;

      case ' ':

        e.preventDefault();

        if (paused) {

          paused = false;

        } else {

          paused = true;

        }

        break;

    }

  });


  // 初始渲染

  draw();

</script>
